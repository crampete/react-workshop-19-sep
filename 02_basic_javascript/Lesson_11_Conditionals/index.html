if and if-else

if (condition) {
// code to be executed when the condition holds true
}
example :
var x = 4;
if (x % 2 == 0) {
console.log("x is even.");
}

if (condition) {
// code to be executed when the condition holds true
} else {
// code to be executed when the condition holds false
}
example :
var x = 5; if (x % 2 == 0) {
console.log("x is even.");
} else {
console.log("x is odd.");
}
